    PROCEDURE GetBalance@1000000000() : Decimal;
    VAR
      "G/L Account"@1000000000 : Record 15;
      DateFilter@1000000001 : Date;
      NBalance@1000000002 : Decimal;
      PBalance@1000000003 : Decimal;
      Balance@1000000004 : Decimal;
    BEGIN
      //Get Beginning Balance
      DateFilter :=CALCDATE('-1D',DateFilterMin) ;
      NBalance := 0 ;
      PBalance := 0 ;

      "G/L Account".RESET ;
      "G/L Account".SETRANGE("G/L Account"."Date Filter",DateFilter) ;
      "G/L Account".SETRANGE("G/L Account"."No.",'21710101','21710109') ;
      IF "G/L Account".FIND('-') THEN
         REPEAT
            //Minus Object
            IF "G/L Account"."No." IN ['21710105','21710106','21710107','21710109'] THEN
               NBalance := NBalance + "G/L Account"."Balance at Date"
            ELSE  //Plus Object
               PBalance := PBalance + "G/L Account"."Balance at Date" ;

         UNTIL "G/L Account".NEXT = 0 ;

      Balance := PBalance - NBalance ;

      EXIT(Balance) ;
    END;

    BEGIN
    END.
