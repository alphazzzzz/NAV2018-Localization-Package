    PROCEDURE ShowCascadeDescription@1000000007(txtPGLAccountNo@1000 : Text;blnPShowDescription2@1001 : Boolean) txtLReturn : Text;
    VAR
      recLGLAccount@1002 : Record 15;
      recxLGLAccount@1005 : Record 15;
      blnLTopLevel@1003 : Boolean;
      intLIndentationLevel@1004 : Integer;
      blnLBreak@1006 : Boolean;
    BEGIN
      txtLReturn := '';

      recLGLAccount.GET(txtPGLAccountNo);

      IF recLGLAccount.Indentation <> 0 THEN BEGIN
        IF blnPShowDescription2 = TRUE THEN BEGIN
          txtLReturn := recLGLAccount."Description CHS";
        END ELSE BEGIN
          txtLReturn := recLGLAccount.Name;
        END;
        blnLBreak := FALSE;
        intLIndentationLevel := recLGLAccount.Indentation;
        REPEAT
          recxLGLAccount.RESET;
          IF recLGLAccount.NEXT(-1) = 0 THEN BEGIN
            blnLBreak := TRUE;
          END ELSE BEGIN
            IF intLIndentationLevel = recLGLAccount.Indentation  + 1 THEN BEGIN
              IF blnPShowDescription2 = TRUE THEN BEGIN
                txtLReturn := recLGLAccount."Description CHS" + '-' + txtLReturn;
              END ELSE BEGIN
                txtLReturn := recLGLAccount.Name + '-' + txtLReturn;
              END;
              intLIndentationLevel := recLGLAccount.Indentation;
            END;
          END;
        UNTIL (recLGLAccount.Indentation = 0) OR (blnLBreak = TRUE);
      END ELSE BEGIN //top level
        IF blnPShowDescription2 = TRUE THEN BEGIN
          txtLReturn := recLGLAccount."Description CHS";
        END ELSE BEGIN
          txtLReturn := recLGLAccount.Name;
        END;
      END;
    END;
